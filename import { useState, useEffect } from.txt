  const [open,     setOpen]     = useState(false);

    window.addEventListener("scroll", fn);
    return () => window.removeEventListener("scroll", fn);
  }, []);
  const links = ["About","Programs","Campus","Research","Admissions","Contact"];
  return (
    <>
      <nav className={scrolled ? "scrolled" : ""}>
        <a href="#" className="nav-logo">Rayat <span>Bahra</span> University</a>
        <ul className="nav-links">
          {links.map(l => <li key={l}><a href={`#${l.toLowerCase()}`}>{l}</a></li>)}

          <button className="nav-cta">Apply Now</button>

            onClick={() => setOpen(o => !o)}

            <span className="ham-line ham-line-1"/>
